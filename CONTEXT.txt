TypeScript

// backend/src/routes/chat.ts
app.post('/api/chat', async (c) => {
  const { message, sessionId, userId } = await c.req.json();
  
  // 1. Rate limit check (1 req/sec per user)
  const lastRequest = await c.env.KV.get(`rate:${userId}`);
  if (lastRequest && Date.now() - parseInt(lastRequest) < 1000) {
    return c.json({ error: 'Rate limit exceeded' }, 429);
  }
  await c.env.KV.put(`rate:${userId}`, Date.now().toString());

  // 2. Retrieve conversation history (last 10 messages for context)
  const history = await c.env.DB.prepare(
    'SELECT role, content FROM messages WHERE sessionId = ? ORDER BY timestamp DESC LIMIT 10'
  ).bind(sessionId).all();

  // 3. Construct Arabic cultural prompt
  const systemPrompt = `أنت معالج نفسي متعاطف... [Cultural context]`;

  // 4. Call Mistral
  const response = await fetch('https://api.mistral.ai/v1/chat/completions', {
    method: 'POST',
    headers: { 'Authorization': `Bearer ${c.env.MISTRAL_API_KEY}` },
    body: JSON.stringify({
      model: 'open-mistral-7b',
      messages: [
        { role: 'system', content: systemPrompt },
        ...history.results,
        { role: 'user', content: message }
      ],
      temperature: 0.7,
      max_tokens: 500
    })
  });

  // 5. Crisis detection (keyword matching)
  const aiResponse = await response.json();
  const content = aiResponse.choices[0].message.content;
  if (containsCrisisKeywords(content + message)) {
    await triggerCrisisProtocol(c.env.DB, sessionId);
  }

  // 6. Store message metadata (content encrypted client-side)
  await c.env.DB.prepare('INSERT INTO messages...').bind(...).run();

  return c.json({ reply: content, sessionId });
});

// src/ai/mistral.ts

interface ChatMessage {
  role: 'system' | 'user' | 'assistant';
  content: string;
}

interface ChatOptions {
  systemPrompt: string;
  messages: ChatMessage[];
  temperature?: number;
  maxTokens?: number;
}

interface ChatResponse {
  content: string;
  usage: { total_tokens: number; prompt_tokens: number; completion_tokens: number };
  model: string;
}

class MistralClient {
  private apiKey: string;
  private baseUrl = 'https://api.mistral.ai/v1';
  
  constructor(apiKey: string) {
    this.apiKey = apiKey;
  }

  async chat(options: ChatOptions): Promise<ChatResponse> {
    const response = await fetch(`${this.baseUrl}/chat/completions`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: 'open-mistral-7b', // Default fast model
        messages: [
          { role: 'system', content: options.systemPrompt },
          ...options.messages,
        ],
        temperature: options.temperature ?? 0.7,
        max_tokens: options.maxTokens ?? 500,
        top_p: 0.9,
        safe_mode: true, // Mistral's safety filter
      }),
    });

    if (!response.ok) {
      if (response.status === 429) throw new RateLimitError('Mistral rate limited');
      if (response.status >= 500) throw new ServiceUnavailableError('Mistral unavailable');
      throw new Error(`Mistral API error: ${response.status}`);
    }

    const data = await response.json();
    
    return {
      content: data.choices[0].message.content,
      usage: data.usage,
      model: data.model,
    };
  }

  // Streaming for future real-time implementation
  async *chatStream(options: ChatOptions): AsyncGenerator<string> {
    const response = await fetch(`${this.baseUrl}/chat/completions`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: 'mistral-small', // Better reasoning for streaming
        messages: [
          { role: 'system', content: options.systemPrompt },
          ...options.messages,
        ],
        temperature: options.temperature ?? 0.7,
        max_tokens: options.maxTokens ?? 500,
        stream: true,
      }),
    });

    const reader = response.body?.getReader();
    if (!reader) throw new Error('No response body');

    const decoder = new TextDecoder();
    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      
      const chunk = decoder.decode(value);
      const lines = chunk.split('\n').filter(line => line.trim() !== '');
      
      for (const line of lines) {
        if (line.startsWith('data: ')) {
          const data = line.slice(6);
          if (data === '[DONE]') return;
          
          try {
            const parsed = JSON.parse(data);
            const content = parsed.choices[0]?.delta?.content;
            if (content) yield content;
          } catch (e) {
            // Skip invalid JSON
          }
        }
      }
    }
  }
}

import { middleware } from 'hono/factory';

export const authMiddleware = middleware(async (c, next) => {
  // Skip auth for health check
  if (c.req.path === '/health') return next();

  const authHeader = c.req.header('Authorization');
  let userId: string | null = null;
  let isAnonymous = true;

  if (authHeader?.startsWith('Bearer ')) {
    const token = authHeader.slice(7);
    
    // Verify JWT (Better Auth or custom)
    try {
      const payload = await verify(token, c.env.JWT_SECRET);
      userId = payload.sub as string;
      isAnonymous = payload.anonymous === true;
    } catch {
      // Invalid token, will create anonymous
    }
  }

  if (!userId) {
    // Generate anonymous user ID
    userId = crypto.randomUUID();
    isAnonymous = true;
    
    // Create user record if not exists
    const db = drizzle(c.env.DB);
    const existing = await db.select().from(users).where(eq(users.id, userId)).get();
    
    if (!existing) {
      await db.insert(users).values({
        id: userId,
        anonymousTag: generateAnonymousTag(),

function generateAnonymousTag(): string {
  const adjectives = ['Seeker', 'Traveler', 'Heart', 'Soul', 'Voice', 'Light'];
  const nouns = ['of Hope', 'in Silence', 'of Dawn', 'under Stars', 'of Courage'];
  const num = Math.floor(Math.random() * 9999);
  const adj = adjectives[Math.floor(Math.random() * adjectives.length)];
  const noun = nouns[Math.floor(Math.random() * nouns.length)];
  return `${adj}_${noun}_${num}`.replace(/\s/g, '');

// src/security/rateLimit.ts

interface RateLimitResult {
  allowed: boolean;
  remaining: number;
  resetAt: number;
  retryAfter?: number;
}

export async function rateLimitCheck(
  kv: KVNamespace,
  userId: string,
  config: { sustained: number; burst: number; window: number }
): Promise<RateLimitResult> {
  const now = Math.floor(Date.now() / 1000);
  const key = `rate:${userId}`;
  
  // Get current window
  const windowStart = Math.floor(now / config.window) * config.window;
  const windowKey = `${key}:${windowStart}`;
  
  const current = await kv.get(windowKey);
  const count = current ? parseInt(current) : 0;
  
  if (count >= config.burst) {
    return {
      allowed: false,
      remaining: 0,
      resetAt: windowStart + config.window,
      retryAfter: windowStart + config.window - now,
    };
  }

  // Check sustained rate (1 req/sec)
  const lastRequestKey = `rate:last:${userId}`;
  const lastRequest = await kv.get(lastRequestKey);
  
  if (lastRequest) {
    const lastTime = parseInt(lastRequest);
    if (now - lastTime < 1) { // Less than 1 second ago
      return {
        allowed: false,
        remaining: Math.max(0, config.burst - count),
        resetAt: windowStart + config.window,
        retryAfter: 1,
      };
    }
  }

  // Allow and increment
  await kv.put(windowKey, (count + 1).toString(), { expirationTtl: config.window });
  await kv.put(lastRequestKey, now.toString(), { expirationTtl: 60 });

  return {
    allowed: true,
    remaining: config.burst - count - 1,
    resetAt: windowStart + config.window,
  };
}

export const rateLimitMiddleware = middleware(async (c, next) => {
  const result = await rateLimitCheck(c.env.KV, c.get('userId'), {
    sustained: 1,
    burst: 10,
    window: 60,
  });

  c.header('X-RateLimit-Limit', '10');
  c.header('X-RateLimit-Remaining', result.remaining.toString());
  c.header('X-RateLimit-Reset', result.resetAt.toString());

  if (!result.allowed) {
    return c.json({ error: 'Rate limit exceeded' }, 429);
  }

  await next();
});

// src/security/crypto.ts

export function generateHash(content: string): string {
  // SHA-256 hash for content integrity
  // In Workers, use Web Crypto API
  const encoder = new TextEncoder();
  const data = encoder.encode(content);
  return crypto.subtle.digest('SHA-256', data).then(buf => 
    Array.from(new Uint8Array(buf))
      .map(b => b.toString(16).padStart(2, '0'))
      .join('')
  );
}

export async function encryptMetadata(data: object, key: CryptoKey): Promise<string> {
  // AES-GCM encryption for sensitive metadata
  const iv = crypto.getRandomValues(new Uint8Array(12));
  const encoder = new TextEncoder();
  const encoded = encoder.encode(JSON.stringify(data));
  
  const encrypted = await crypto.subtle.encrypt(
    { name: 'AES-GCM', iv },
    key,
    encoded
  );
  
  // Return IV + ciphertext as base64
  const combined = new Uint8Array(iv.length + encrypted.byteLength);
  combined.set(iv);
  combined.set(new Uint8Array(encrypted), iv.length);
  
  return btoa(String.fromCharCode(...combined));
}

export const localeMiddleware = middleware(async (c, next) => {
  // Priority: 1) Header, 2) Query param, 3) GeoIP, 4) Default
  const headerLang = c.req.header('Accept-Language');
  const queryLang = c.req.query('lang');
  
  let locale = 'ar'; // Default
  let region = 'tunisia'; // Default
  
  if (queryLang && ['ar', 'ar-TN', 'fr', 'en'].includes(queryLang)) {
    locale = queryLang;
  } else if (headerLang) {
    // Parse Accept-Language
    const parsed = parseAcceptLanguage(headerLang);
    if (parsed.includes('ar-TN')) locale = 'ar-TN';
    else if (parsed.includes('ar')) locale = 'ar';
    else if (parsed.includes('fr')) locale = 'fr';
    else if (parsed.includes('en')) locale = 'en';
  }
  
  // GeoIP region detection from Cloudflare
  const cfCountry = c.req.raw.cf?.country;
  if (cfCountry) {
    const regionMap: Record<string, string> = {
      'TN': 'tunisia',
      'SA': 'gcc', 'AE': 'gcc', 'QA': 'gcc', 'KW': 'gcc', 'BH': 'gcc', 'OM': 'gcc',
      'EG': 'egypt',
      'LB': 'levant', 'SY': 'levant', 'JO': 'levant', 'PS': 'levant', 'IQ': 'levant',
    };
    region = regionMap[cfCountry] || 'diaspora';
  }
  
  c.set('preferredLanguage', locale);
  c.set('region', region);
  c.set('isRTL', locale.startsWith('ar'));
  
  await next();
});

webhookRoutes.post('/cron/data-cleanup', async (c) => {

  const cutoff = new Date();
  cutoff.setDate(cutoff.getDate() - 30);
  
  const toDelete = await db
    .select({ id: users.id })
    .from(users)
    .where(
      and(
        eq(users.isDeleted, true),
        sql`${users.deletionRequestedAt} < ${cutoff}`
      )
    );
  
  for (const user of toDelete) {
    // Cascade delete all related data
    await db.delete(messages).where(eq(messages.sessionId, 
      db.select({ id: sessions.id }).from(sessions).where(eq(sessions.userId, user.id))
    ));
    await db.delete(sessions).where(eq(sessions.userId, user.id));
    await db.delete(moodEntries).where(eq(moodEntries.userId, user.id));
    await db.delete(crisisInterventions).where(eq(crisisInterventions.sessionId,
      db.select({ id: sessions.id }).from(sessions).where(eq(sessions.userId, user.id))
    ));
    await db.delete(users).where(eq(users.id, user.id));
  }
  
  return c.json({ deleted: toDelete.length });

8. DATA & PRIVACY MODEL
8.1 Zero-Knowledge Architecture
Table

Message metadata	Cloudflare D1	None (hashes only)	Read-only

Mood data	D1 (anonymized)	None (aggregate only)	Read/write
Crisis logs	D1	None (compliance requirement)	Read/write
8.2 Encryption Specification
Algorithm: AES-256-GCM
Key derivation: PBKDF2 (100,000 iterations) from device ID + optional user passphrase
IV: 12-byte random per message
Authentication: GCM tag included
Key rotation: Automatic monthly, re-encrypt on access
Client-Side Flow:
plain
Copy
1. User types message
2. Message encrypted with current key
3. SHA-256 hash of ciphertext generated
4. Ciphertext stored in IndexedDB/SQLite
5. Metadata {hash, length, iv, timestamp} sent to server
6. Server stores metadata only
7. Ciphertext synced to R2 (backup) if user opts in
8.3 Privacy Guarantees
We cannot decrypt your messages — keys never leave your device
We cannot identify you — anonymous by default, optional email
We delete everything on request — 30-day grace, then hard delete
We share nothing — no data sales, no ad targeting, no "partners"
We comply legally — minimal data for valid legal requests (metadata only)
